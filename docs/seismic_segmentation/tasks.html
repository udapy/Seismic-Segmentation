 42    tuning_dir.mkdir(parents=True, exist_ok=True)

 44    # Set up logger
 45    logger = get_logger(str(tuning_dir))

 48
 49    # Save config
 50    with open(tuning_dir / 'config.yaml', 'w') as f:
 51        yaml.dump(config, f, default_flow_style=False)
 52
 53    # Define objective function
 54    def objective(trial: Trial):
 55        # Set up trial directory
 56        trial_dir = tuning_dir / f"trial_{trial.number}"
 57        trial_dir.mkdir(parents=True, exist_ok=True)
 58
 59        # Get hyperparameters
 60        trial_config = config.copy()
 61
 62        # Override with trial-specific hyperparameters
 63        params = config.get('params', {})
 64        for param_name, param_config in params.items():
 65            param_type = param_config.get('type', 'float')
 66
 67            if param_type == 'float':
 68                low = param_config.get('low', 1e-5)
 69                high = param_config.get('high', 1.0)
 70                log = param_config.get('log', False)
 71                value = trial.suggest_float(param_name, low, high, log=log)
 72            elif param_type == 'int':
 73                low = param_config.get('low', 1)
 74                high = param_config.get('high', 100)
 75                value = trial.suggest_int(param_name, low, high)
 76            elif param_type == 'categorical':
 77                choices = param_config.get('choices', [])
 78                value = trial.suggest_categorical(param_name, choices)
 79
 80            trial_config[param_name] = value
 81
 82        # Create data loaders
 83        train_loader = get_dataloader(trial_config, split='train')
 84        val_loader = get_dataloader(trial_config, split='val')
 85
 86        # Create model
 87        model = get_model(trial_config)
 88        model = model.to(device)
 89
 90        # Create loss function
 91        criterion = get_loss_fn(trial_config)
 92
 93        # Create optimizer
 94        lr = trial_config.get('lr', 0.001)
 95        weight_decay = trial_config.get('weight_decay', 0.0001)
 96
 97        if trial_config.get('optimizer', 'adamw').lower() == 'sgd':
 98            optimizer = torch.optim.SGD(model.parameters(), lr=lr, momentum=0.9, weight_decay=weight_decay)
 99        else:
100            optimizer = torch.optim.AdamW(model.parameters(), lr=lr, weight_decay=weight_decay)
101
102        # Create scheduler
103        scheduler_type = trial_config.get('scheduler', 'cosine')
104        if scheduler_type == 'plateau':
105            scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
106                optimizer, mode='min', factor=0.1, patience=5, verbose=True
107            )
108        elif scheduler_type == 'cosine':
109            epochs = trial_config.get('epochs', 10)  # Use fewer epochs for tuning
110            scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
111                optimizer, T_max=epochs, eta_min=1e-6
112            )
113        else:
114            scheduler = None
115
116        # Create trainer
117        trainer = Trainer(
118            model=model,
119            train_loader=train_loader,
120            val_loader=val_loader,
121            optimizer=optimizer,
122            scheduler=scheduler,
123            criterion=criterion,
124            device=device,
125            config=trial_config,
126            experiment_dir=trial_dir
127        )
128
129        # Run training
130        trainer.train()
131
132        # Get best validation metric
133        metric_name = config.get('metric', 'val_mean_iou')
134        if metric_name == 'val_loss':
135            metric_value = min(trainer.val_losses) if trainer.val_losses else float('inf')
136        elif metric_name == 'val_mean_iou':
137            metric_value = max(trainer.val_metrics_history['mean_iou']) if trainer.val_metrics_history['mean_iou'] else 0.0
138        elif metric_name == 'val_accuracy':
139            metric_value = max(trainer.val_metrics_history['accuracy']) if trainer.val_metrics_history['accuracy'] else 0.0
140        elif metric_name == 'val_f1_score':
141            metric_value = max(trainer.val_metrics_history['f1_score']) if trainer.val_metrics_history['f1_score'] else 0.0
142
143        return metric_value
144
145    try:
146        # Create study
147        direction = config.get('direction', 'maximize')
148        study_name = experiment_name
149        storage_name = f"sqlite:///{tuning_dir}/study.db"
150
151        study = optuna.create_study(

 15def run(config):
 16    """
 17    Run model promotion task.
 18    
 19    Args:
 20        config: Configuration dictionary
 21    
 22    Returns:
 23        True if successful, False otherwise
 24    """
 25    # Set up directories
 26    models_dir = Path(config.get('models_dir', 'output'))
 27    output_dir = Path(config.get('output_dir', 'models/production'))
 28    output_dir.mkdir(parents=True, exist_ok=True)
 29
 30    # Set up logger
 31    logger = get_logger(str(output_dir))
 32    logger.info(f"Starting model promotion")
 33
 34    try:
 35        # Find all models
 36        model_dirs = []
 37        for item in models_dir.glob('*'):
 38            if item.is_dir() and (item / 'checkpoints' / 'best_model.pth').exists():
 39                model_dirs.append(item)
 40
 41        if not model_dirs:
 42            logger.error(f"No models found in {models_dir}")
 43            return False
 44
 45        logger.info(f"Found {len(model_dirs)} model(s)")
 46
 47        # Evaluate all models
 48        model_metrics = []
 49        for model_dir in model_dirs:
 50            checkpoint_path = model_dir / 'checkpoints' / 'best_model.pth'
 51            config_path = model_dir / 'config.yaml'
 52
 53            # Load checkpoint
 54            checkpoint = torch.load(checkpoint_path, map_location='cpu')
 55
 56            # Load config
 57            if config_path.exists():
 58                with open(config_path, 'r') as f:
 59                    model_config = yaml.safe_load(f)
 60            else:
 61                model_config = checkpoint.get('config', {})
 62
 63            # Get metrics
 64            metrics = {}
 65            if 'loss' in checkpoint:
 66                metrics['loss'] = checkpoint['loss']
 67
 68            # Check if evaluation results exist
 69            eval_results_path = model_dir / 'evaluations' / 'results.json'
 70            if eval_results_path.exists():
 71                with open(eval_results_path, 'r') as f:
 72                    eval_results = json.load(f)
 73                metrics.update(eval_results.get('overall_metrics', {}))
 74
 75            model_metrics.append({
 76                'dir': str(model_dir),
 77                'name': model_dir.name,
 78                'checkpoint': str(checkpoint_path),
 79                'config': model_config,
 80                'metrics': metrics
 81            })
 82
 83        # Sort models by primary metric
 84        primary_metric = config.get('primary_metric', 'mean_iou')
 85        threshold = config.get('threshold', {}).get(primary_metric, 0.0)
 86
 87        # Filter models by threshold
 88        qualified_models = [
 89            model for model in model_metrics
 90            if model['metrics'].get(primary_metric, 0.0) >= threshold
 91        ]
 92
 93        if not qualified_models:
 94            logger.error(f"No models meet the threshold ({threshold}) for {primary_metric}")
 95            return False
 96
 97        # Sort by primary metric (descending)
 98        qualified_models.sort(key=lambda x: x['metrics'].get(primary_metric, 0.0), reverse=True)
 99
100        # Get best model
101        best_model = qualified_models[0]
102        logger.info(f"Selected best model: {best_model['name']} with {primary_metric}={best_model['metrics'].get(primary_metric, 0.0):.4f}")
103
104        # Create version
105        if config.get('version', 'auto') == 'auto':
106            version = datetime.now().strftime("%Y%m%d_%H%M%S")
107        else:
108            version = config.get('version')
109
110        # Create version directory
111        version_dir = output_dir / version
112        version_dir.mkdir(parents=True, exist_ok=True)
113
114        # Copy model files
115        shutil.copy2(best_model['checkpoint'], version_dir / 'model.pth')
116
117        # Save model info
118        model_info = {
119            'version': version,
120            'source': best_model['dir'],
121            'name': best_model['name'],
122            'metrics': best_model['metrics'],
123            'config': best_model['config'],
124            'promoted_at': datetime.now().isoformat()
125        }
126
127        with open(version_dir / 'model_info.json', 'w') as f:
128            json.dump(model_info, f, indent=2)
129
130        # Create latest symlink
131        latest_link = output_dir / 'latest'
132        if latest_link.exists():
133            if latest_link.is_symlink():
134                latest_link.unlink()
135            else:
136                shutil.rmtree(latest_link)
137
138        # Create relative symlink
139        os.symlink(version, latest_link, target_is_directory=True)
140
141        logger.info(f"Model promoted successfully to {version_dir}")
142        logger.info(f"Created 'latest' symlink to {version}")
143
144        return True
145
146    except Exception as e:
147        logger.error(f"Error during model promotion: {e}")
148        if config.get('debug', False):
149            import traceback
150            logger.error(traceback.format_exc())
151        return False

 86    def train(self):
 87        """Train model for specified number of epochs."""
 88        self.logger.info(f"Starting training with {self.epochs} epochs")
 89
 90        for epoch in range(self.epochs):
 91            # Train for one epoch
 92            train_loss = self.train_one_epoch(epoch)
 93
 94            # Validate
 95            val_loss, val_metrics = self.validate(epoch)
 96
 97            # Update scheduler
 98            if hasattr(self.scheduler, 'step'):
 99                if isinstance(self.scheduler, torch.optim.lr_scheduler.ReduceLROnPlateau):
100                    self.scheduler.step(val_loss)
101                else:
102                    self.scheduler.step()
103
104            # Log metrics
105            self.logger.info(f'Epoch: {epoch+1}/{self.epochs}, '
106                           f'Train Loss: {train_loss:.4f}, '
107                           f'Val Loss: {val_loss:.4f}, '
108                           f'Mean IoU: {val_metrics["mean_iou"]:.4f}, '
109                           f'Accuracy: {val_metrics["accuracy"]:.4f}')
110
111            # Update tracking
112            self.train_losses.append(train_loss)
113            self.val_losses.append(val_loss)
114            self.val_metrics_history['mean_iou'].append(val_metrics['mean_iou'])
115            self.val_metrics_history['accuracy'].append(val_metrics['accuracy'])
116            self.val_metrics_history['f1_score'].append(val_metrics['f1_score'])
117
118            # Save model if it's the best so far
119            if val_loss < self.best_val_loss:
120                self.best_val_loss = val_loss
121                self.save_checkpoint(epoch, is_best=True)
122                self.logger.info(f'Saved best model with validation loss: {val_loss:.4f}')
123
124            # Save checkpoint at regular intervals
125            if (epoch + 1) % self.save_interval == 0:
126                self.save_checkpoint(epoch)
127
128            # Write to TensorBoard
129            self.writer.add_scalar('Loss/train', train_loss, epoch)
130            self.writer.add_scalar('Loss/val', val_loss, epoch)
131            for metric_name, metric_value in val_metrics.items():
132                self.writer.add_scalar(f'Metrics/{metric_name}', metric_value, epoch)
133
134        self.writer.close()
135        self.logger.info("Training complete!")

137    def train_one_epoch(self, epoch):
138        """Train for one epoch."""
139        self.model.train()
140        running_loss = 0.0
141
142        pbar = tqdm(self.train_loader, desc=f'Epoch {epoch+1}/{self.epochs}')
143        for batch_idx, batch in enumerate(pbar):
144            # Get data
145            images = batch['image'].to(self.device)
146            masks = batch['mask'].to(self.device)
147
148            # Zero the parameter gradients
149            self.optimizer.zero_grad()
150
151            # Forward
152            if hasattr(self.model, 'use_sag') and self.model.use_sag:
153                # For SAG model, generate dummy prompts
154                B = images.size(0)
155                point_prompts = torch.rand(B, 1, 3, device=self.device)
156                box_prompts = torch.rand(B, 1, 4, device=self.device)
157                well_prompts = torch.rand(B, 1, 10, device=self.device)
158
159                outputs = self.model(
160                    images,
161                    point_prompts=point_prompts,
162                    box_prompts=box_prompts,
163                    well_prompts=well_prompts
164                )
165            else:
166                outputs = self.model(images)
167
168            # Calculate loss
169            loss = self.criterion(outputs, masks)
170
171            # Backward and optimize
172            loss.backward()
173            self.optimizer.step()
174
175            # Update running loss
176            running_loss += loss.item() * images.size(0)
177
178            # Update progress bar
179            pbar.set_postfix({'loss': loss.item()})
180
181            # Log batch loss to TensorBoard
182            global_step = epoch * len(self.train_loader) + batch_idx
183            self.writer.add_scalar('Loss/train_step', loss.item(), global_step)
184
185            # Visualize predictions
186            if (batch_idx + 1) % self.visualize_interval == 0:
187                self.visualize_predictions(images, masks, outputs, epoch, batch_idx, split='train')
188
189        epoch_loss = running_loss / len(self.train_loader.dataset)
190        return epoch_loss

192    def validate(self, epoch):
193        """Validate the model."""
194        self.model.eval()
195        val_loss = 0.0
196        all_preds = []
197        all_masks = []
198
199        with torch.no_grad():
200            for batch_idx, batch in enumerate(tqdm(self.val_loader, desc='Validating')):
201                # Get data
202                images = batch['image'].to(self.device)
203                masks = batch['mask'].to(self.device)
204
205                # Forward
206                if hasattr(self.model, 'use_sag') and self.model.use_sag:
207                    # For SAG model, generate dummy prompts
208                    B = images.size(0)
209                    point_prompts = torch.rand(B, 1, 3, device=self.device)
210                    box_prompts = torch.rand(B, 1, 4, device=self.device)
211                    well_prompts = torch.rand(B, 1, 10, device=self.device)
212
213                    outputs = self.model(
214                        images,
215                        point_prompts=point_prompts,
216                        box_prompts=box_prompts,
217                        well_prompts=well_prompts
218                    )
219                else:
220                    outputs = self.model(images)
221
222                # Calculate loss
223                loss = self.criterion(outputs, masks)
224                val_loss += loss.item() * images.size(0)
225
226                # Get predictions
227                if self.config.get('n_classes', 1) > 1:
228                    preds = torch.argmax(outputs, dim=1)
229                else:
230                    preds = (torch.sigmoid(outputs) > 0.5).float()
231
232                # Store predictions and masks for metrics calculation
233                all_preds.append(preds.cpu().numpy())
234                all_masks.append(masks.cpu().numpy())
235
236                # Visualize first batch
237                if batch_idx == 0:
238                    self.visualize_predictions(images, masks, outputs, epoch, batch_idx, split='val')
239
240        # Calculate epoch loss
241        epoch_loss = val_loss / len(self.val_loader.dataset)
242
243        # Calculate metrics
244        all_preds = np.concatenate([p.flatten() for p in all_preds])
245        all_masks = np.concatenate([m.flatten() for m in all_masks])
246        metrics = get_metrics(all_masks, all_preds, self.config.get('n_classes', 1))
247
248        return epoch_loss, metrics

250    def visualize_predictions(self, images, masks, outputs, epoch, batch_idx, split='train'):
251        """Visualize model predictions."""
252        if self.config.get('n_classes', 1) > 1:
253            preds = torch.argmax(outputs, dim=1)

267    def save_checkpoint(self, epoch, is_best=False):
268        """Save model checkpoint."""
269        checkpoint = {
270            'epoch': epoch,
271            'model_state_dict': self.model.state_dict(),
272            'optimizer_state_dict': self.optimizer.state_dict(),
273            'scheduler_state_dict': self.scheduler.state_dict() if hasattr(self.scheduler, 'state_dict') else None,
274            'loss': self.val_losses[-1] if self.val_losses else float('inf'),
275            'config': self.config
276        }
277
278        if is_best:
279            checkpoint_path = self.checkpoints_dir / 'best_model.pth'
280        else:
281            checkpoint_path = self.checkpoints_dir / f'checkpoint_epoch{epoch+1}.pth'
282
283        torch.save(checkpoint, checkpoint_path)
